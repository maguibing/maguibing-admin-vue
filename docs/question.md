## 高级布局组件 `npm install vue-grid-layout --save`

::: tip
中文文档
https://github.com/jbaysolutions/vue-grid-layout/blob/HEAD/README-zh_CN.md
:::

```vue
<grid-layout
  :layout.sync="layout"
  :col-num="12"
  :row-height="30"
  :is-draggable="true"
  :is-resizable="true"
  :is-mirrored="false"
  :vertical-compact="true"
  :margin="[10, 10]"
  :use-css-transforms="true"
>
  <grid-item v-for="item in layout"
    :x="item.x"
    :y="item.y"
    :w="item.w"
    :h="item.h"
    :i="item.i"
    :key="item.i">
      {{item.i}}
  </grid-item>
</grid-layout>
```
